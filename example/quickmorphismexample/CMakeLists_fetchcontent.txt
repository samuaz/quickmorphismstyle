cmake_minimum_required(VERSION 3.21)

project(quickmorphismexample 
    VERSION 1.0.0 
    LANGUAGES CXX
    DESCRIPTION "QuickMorphism Style Example Application with FetchContent"
)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Qt6 setup
set(CMAKE_INCLUDE_CURRENT_DIR ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# Find Qt6 components
find_package(Qt6 6.2 REQUIRED COMPONENTS 
    Core 
    Quick 
    Gui 
    Qml 
    QuickControls2
)

# Qt6 setup
qt_standard_project_setup()

# FetchContent to get QuickMorphismStyle from repository
include(FetchContent)

FetchContent_Declare(
    QuickMorphism
    GIT_REPOSITORY https://github.com/samuaz/quickmorphismstyle.git
    GIT_TAG qt6.9-migration  # Use the Qt6.9 migration branch
    GIT_PROGRESS TRUE
)

# Make QuickMorphism available
FetchContent_MakeAvailable(QuickMorphism)

# Create executable
qt_add_executable(appquickmorphismexample
    main.cpp
)

# Add QML module for the example
qt_add_qml_module(appquickmorphismexample
    URI quickmorphismexample
    VERSION ${PROJECT_VERSION}
    QML_FILES 
        demo.qml 
        MainForm.ui.qml
)

# Target properties
set_target_properties(appquickmorphismexample PROPERTIES
    MACOSX_BUNDLE_GUI_IDENTIFIER com.theopencompany.quickmorphismexample
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

# Link libraries
target_link_libraries(appquickmorphismexample PRIVATE
    Qt6::Core
    Qt6::Quick
    Qt6::Gui
    Qt6::Qml
    Qt6::QuickControls2
    QuickMorphism
)

# Compile definitions
target_compile_definitions(appquickmorphismexample PRIVATE
    $<$<OR:$<CONFIG:Debug>,$<CONFIG:RelWithDebInfo>>:QT_QML_DEBUG>
    QT_DEPRECATED_WARNINGS
)

# Installation
include(GNUInstallDirs)

install(TARGETS appquickmorphismexample
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

# Set QML import path for Qt Creator
get_target_property(QuickMorphism_BINARY_DIR QuickMorphism BINARY_DIR)
set(QML_IMPORT_PATH 
    "${QuickMorphism_BINARY_DIR}"
    "${CMAKE_BINARY_DIR}"
    CACHE STRING "Qt Creator import paths" FORCE
)

# Debug output
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
    message(STATUS "QML import paths: ${QML_IMPORT_PATH}")
    message(STATUS "QuickMorphism binary dir: ${QuickMorphism_BINARY_DIR}")
endif()
