cmake_minimum_required(VERSION 3.21)

project(quickmorphismexample 
    VERSION 1.0.0 
    LANGUAGES CXX
    DESCRIPTION "QuickMorphism Style Example Application - Local Development"
)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Qt6 setup
set(CMAKE_INCLUDE_CURRENT_DIR ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# Find Qt6 components
find_package(Qt6 6.2 REQUIRED COMPONENTS 
    Core 
    Quick 
    Gui 
    Qml 
    QuickControls2
)

# Qt6 setup
qt_standard_project_setup()

# ===================================
# Determinar método de inclusión
# ===================================

# Para desarrollo: intentar usar repositorio local primero
set(LOCAL_QUICKMORPHISM_PATH "${CMAKE_CURRENT_SOURCE_DIR}/../../")

if(EXISTS "${LOCAL_QUICKMORPHISM_PATH}/CMakeLists.txt")
    # Desarrollo local: usar add_subdirectory
    message(STATUS "Using local QuickMorphism development version from: ${LOCAL_QUICKMORPHISM_PATH}")
    set(QUICKMORPHISM_SOURCE_DIR ${LOCAL_QUICKMORPHISM_PATH})
    add_subdirectory(${LOCAL_QUICKMORPHISM_PATH} quickmorphism_build)
    set(QUICKMORPHISM_METHOD "local")
else()
    # Producción: usar FetchContent
    message(STATUS "Local development version not found, using FetchContent...")
    include(FetchContent)
    
    FetchContent_Declare(
        QuickMorphism
        GIT_REPOSITORY https://github.com/samuaz/quickmorphismstyle.git
        GIT_TAG qt6.9-migration
    )
    
    FetchContent_MakeAvailable(QuickMorphism)
    FetchContent_GetProperties(QuickMorphism)
    set(QUICKMORPHISM_SOURCE_DIR ${quickmorphism_SOURCE_DIR})
    set(QUICKMORPHISM_METHOD "fetchcontent")
endif()

# Create executable
qt_add_executable(appquickmorphismexample
    main.cpp
)

# Add QML module for the example
qt_add_qml_module(appquickmorphismexample
    URI quickmorphismexample
    VERSION ${PROJECT_VERSION}
    NO_GENERATE_PLUGIN_SOURCE
    QML_FILES 
        Demo.qml 
        MainForm.ui.qml
)

# Target properties
set_target_properties(appquickmorphismexample PROPERTIES
    MACOSX_BUNDLE_GUI_IDENTIFIER com.theopencompany.quickmorphismexample.${QUICKMORPHISM_METHOD}
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

# Link libraries
target_link_libraries(appquickmorphismexample PRIVATE
    Qt6::Core
    Qt6::Quick
    Qt6::Gui
    Qt6::Qml
    Qt6::QuickControls2
    QuickMorphism
)

# Include directories for headers
target_include_directories(appquickmorphismexample PRIVATE
    ${QUICKMORPHISM_SOURCE_DIR}/include
)

# ===================================
# QML Import Path Configuration
# ===================================

# Configure paths based on method
set(QUICKMORPHISM_QML_SOURCE "${QUICKMORPHISM_SOURCE_DIR}/QuickMorphism")

if(QUICKMORPHISM_METHOD STREQUAL "local")
    set(QUICKMORPHISM_BUILD_OUTPUT "${CMAKE_BINARY_DIR}/quickmorphism_build")
else()
    set(QUICKMORPHISM_BUILD_OUTPUT "${quickmorphism_BINARY_DIR}")
endif()

# Set QML import paths
set(QML_IMPORT_PATH 
    "${QUICKMORPHISM_QML_SOURCE}"          # Source QML files
    "${QUICKMORPHISM_BUILD_OUTPUT}"        # Build output  
    "${CMAKE_CURRENT_BINARY_DIR}"          # Current build
    "${CMAKE_CURRENT_SOURCE_DIR}"          # Current source
    CACHE STRING "Qt Creator import paths" FORCE
)

# Copy QuickMorphism QML files for Qt Design Studio
add_custom_command(TARGET appquickmorphismexample POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${QUICKMORPHISM_SOURCE_DIR}/QuickMorphism"
        "${CMAKE_CURRENT_BINARY_DIR}/QuickMorphism"
    COMMENT "Copying QuickMorphism QML files (${QUICKMORPHISM_METHOD}) for Qt Design Studio"
)

# Compile definitions
target_compile_definitions(appquickmorphismexample PRIVATE
    $<$<OR:$<CONFIG:Debug>,$<CONFIG:RelWithDebInfo>>:QT_QML_DEBUG>
    QT_DEPRECATED_WARNINGS
    QUICKMORPHISM_${QUICKMORPHISM_METHOD}_BUILD
)

# Installation
include(GNUInstallDirs)

install(TARGETS appquickmorphismexample
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

# Debug output
message(STATUS "=== QuickMorphism Integration Configuration ===")
message(STATUS "Integration method: ${QUICKMORPHISM_METHOD}")
message(STATUS "QuickMorphism source: ${QUICKMORPHISM_SOURCE_DIR}")
message(STATUS "QML source path: ${QUICKMORPHISM_QML_SOURCE}")
message(STATUS "QML build output: ${QUICKMORPHISM_BUILD_OUTPUT}")
message(STATUS "QML import paths: ${QML_IMPORT_PATH}")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "==============================================")
