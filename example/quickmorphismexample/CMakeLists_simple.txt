cmake_minimum_required(VERSION 3.21)

project(quickmorphismexample 
    VERSION 1.0.0 
    LANGUAGES CXX
    DESCRIPTION "QuickMorphism Style Example Application - Simple Version"
)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Qt6 setup
qt_standard_project_setup()

# Find Qt6 components
find_package(Qt6 6.2 REQUIRED COMPONENTS 
    Core Quick Gui Qml QuickControls2
)

# Create executable
qt_add_executable(appquickmorphismexample main.cpp)

# Add QML module for the example
qt_add_qml_module(appquickmorphismexample
    URI quickmorphismexample
    VERSION ${PROJECT_VERSION}
    QML_FILES demo.qml MainForm.ui.qml
    RESOURCES 
        ../../QuickMorphism/assets/font/fa-brands-400.ttf
        ../../QuickMorphism/assets/font/fa-regular-400.ttf
        ../../QuickMorphism/assets/font/fa-solid-900.ttf
        ../../QuickMorphism/assets/font/NunitoSans-Regular.ttf
)

# Add QuickMorphism QML files as resources
qt_add_resources(appquickmorphismexample "quickmorphism_qml"
    PREFIX "/"
    BASE "../../"
    FILES
        QuickMorphism/ApplicationWindow.qml
        QuickMorphism/Button.qml
        QuickMorphism/CheckBox.qml
        QuickMorphism/Config.qml
        QuickMorphism/Label.qml
        QuickMorphism/Page.qml
        QuickMorphism/QuickMorphismUI.qml
        QuickMorphism/Switch.qml
        QuickMorphism/TextField.qml
        QuickMorphism/ToolBar.qml
        QuickMorphism/ToolButton.qml
        QuickMorphism/SnackBar.qml
        QuickMorphism/RoundImage.qml
        QuickMorphism/ScrollableContainer.qml
        QuickMorphism/themes/QuickMorphismDarkTheme.qml
        QuickMorphism/themes/QuickMorphismLightTheme.qml
        QuickMorphism/def/DirectoryFontLoader.qml
        QuickMorphism/qmldir
        QuickMorphism/themes/qmldir
        QuickMorphism/def/qmldir
)

# Target properties
set_target_properties(appquickmorphismexample PROPERTIES
    MACOSX_BUNDLE_GUI_IDENTIFIER com.theopencompany.quickmorphismexample
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

# Link libraries
target_link_libraries(appquickmorphismexample PRIVATE
    Qt6::Core
    Qt6::Quick
    Qt6::Gui
    Qt6::Qml
    Qt6::QuickControls2
)

# Add include directories
target_include_directories(appquickmorphismexample PRIVATE
    "../../include"
)

# Add QuickMorphism source files directly
target_sources(appquickmorphismexample PRIVATE
    ../../src/quickmorphism.cpp
    ../../src/ui/quickmorphismui.cpp
    ../../src/ui/snackbar.cpp
    ../../src/statusbar/statusbar.cpp
    ../../src/statusbar/statusbar_dummy.cpp
)

# Compile definitions
target_compile_definitions(appquickmorphismexample PRIVATE
    $<$<OR:$<CONFIG:Debug>,$<CONFIG:RelWithDebInfo>>:QT_QML_DEBUG>
)
