cmake_minimum_required(VERSION 3.21)

project(QuickMorphismStyleTest VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 6.2 REQUIRED COMPONENTS Core Quick QuickControls2)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

qt_add_executable(QuickMorphismStyleTest
    test_quickmorphism_style.cpp
    test_style.qrc
)

target_link_libraries(QuickMorphismStyleTest
    Qt6::Core
    Qt6::Quick
    Qt6::QuickControls2
)

# Set QML import path to find the style
set_property(TARGET QuickMorphismStyleTest PROPERTY QT_QML_IMPORT_PATH "${CMAKE_BINARY_DIR}/qml")
